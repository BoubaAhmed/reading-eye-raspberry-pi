    \begin{lstlisting}[language=bash]
ssh pi@192.168.43.197

    \end{lstlisting}
    
    \textbf{Étape 3 : Installer les dépendances système}
    \begin{lstlisting}[language=bash]
sudo bash system_setup.sh
sudo reboot
    \end{lstlisting}
